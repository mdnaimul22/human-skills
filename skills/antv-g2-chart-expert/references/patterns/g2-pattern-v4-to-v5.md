## Change 1: Import Method

```javascript
// ❌ G2 v4 Syntax
import G2 from '@antv/g2';
const chart = new G2.Chart({ container: 'container' });

// ✅ G2 v5 Syntax
import { Chart } from '@antv/g2';
const chart = new Chart({ container: 'container' });
```

## Change 2: Data Binding

```javascript
// ❌ G2 v4: Separate call to .data()
chart.data(data);

// ✅ G2 v5 (Method 1): Pass on the mark
chart.options({ type: 'interval', data, encode: {...} });

// ✅ G2 v5 (Method 2): Data field in options
chart.options({
  type: 'interval',
  data,    // data as a field in options
  encode: { x: 'month', y: 'value' },
});
```

## Change 3: Field Encoding API

```javascript
// ❌ G2 v4: Chained .encode() method
chart.interval()
  .encode('x', 'month')
  .encode('y', 'value')
  .encode('color', 'product');

// ✅ G2 v5: encode as an object field
chart.options({
  type: 'interval',
  encode: {
    x: 'month',
    y: 'value',
    color: 'product',
  },
});
```

## Change 4: Data Transformation (transform)

```javascript
// ❌ G2 v4: Different method names and parameters
chart.interval().adjust('stack');     // Stack
chart.interval().adjust('dodge');     // Dodge

// ✅ G2 v5: Use transform array
chart.options({
  type: 'interval',
  transform: [{ type: 'stackY' }],   // Stack
  // transform: [{ type: 'dodgeX' }],  // Dodge
});
```

## Change 5: Coordinate System Configuration

```javascript
// ❌ G2 v4
chart.coordinate('polar');
chart.coordinate('theta');
chart.coordinate().transpose();

// ✅ G2 v5: Configure via the coordinate field object
chart.options({
  coordinate: { type: 'polar' },
  // coordinate: { type: 'theta' },
  // coordinate: { transform: [{ type: 'transpose' }] },
});
```

## Change 6: Auxiliary Annotation (guide → annotation)

```javascript
// ❌ G2 v4: guide API
chart.guide().line({ start: ['min', 50], end: ['max', 50] });
chart.guide().text({ position: ['median', 'median'], content: '中位线' });

// ✅ G2 v5: Use lineY/lineX mark (in view + children)
chart.options({
  type: 'view',
  data,
  children: [
    { type: 'line', encode: { x: 'month', y: 'value' } },
    {
      type: 'lineY',
      data: [{ threshold: 50 }],
      encode: { y: 'threshold' },
      style: { stroke: 'red', lineDash: [4, 4] },
      labels: [{ text: 'Target Line', position: 'right' }],
    },
  ],
});
```

## Change 7: Label Configuration

```javascript
// ❌ G2 v4: .label() method
chart.interval().label('value');
chart.interval().label({ fields: ['value'], formatter: (v) => `${v}万` });

// ✅ G2 v5: labels array (note the plural)
chart.options({
  type: 'interval',
  labels: [
    {
      text: 'value',
      // or: text: (d) => `${d.value}万`,
    },
  ],
});
```
## Change 8: Tooltip Configuration

```javascript
// ❌ G2 v4
chart.tooltip({
  showTitle: false,
  itemTpl: '<li>{name}: {value}</li>',
});

// ✅ G2 v5
chart.options({
  tooltip: {
    title: (d) => d.month,
    items: [
      { field: 'value', name: 'Value', valueFormatter: (v) => `${v}k` },
    ],
  },
  interaction: [{ type: 'tooltip' }],
});
```

## Change 9: Multiple Mark Overlay

```javascript
// ❌ G2 v4: Directly overlay by calling chart method multiple times
chart.line().encode('x', 'month').encode('y', 'value');
chart.point().encode('x', 'month').encode('y', 'value');

// ✅ G2 v5: Use type: 'view' + children
chart.options({
  type: 'view',
  data,
  encode: { x: 'month', y: 'value' },
  children: [
    { type: 'line' },
    { type: 'point' },
  ],
});
```

## Change 10: Animation Configuration

```javascript
// ❌ G2 v4
chart.animate(true);

// ✅ G2 v5
chart.options({
  animate: {
    enter: { type: 'fadeIn', duration: 300 },
    update: { type: 'morphing', duration: 500 },
  },
});
```

## v4 Error Pattern Quick Check

When reviewing LLM-generated code, check for the following common v4 error patterns:

- [ ] `import G2 from '@antv/g2'` (should be a named import)
- [ ] `chart.source(data)` or `chart.data(data)` (should be in options)
- [ ] `.position('x*y')` (v4 syntax)
- [ ] `.adjust('stack')` / `.adjust('dodge')` (should be a transform array)
- [ ] `chart.guide().line()` (should be lineY/lineX mark)
- [ ] `.label()` method (should be a `labels` array)
- [ ] Multiple `chart.options()` calls (apply view + children)
